CREATE DATABASE asribank;

CREATE USER 'asriuser'@'%' IDENTIFIED BY 'asripassword';
GRANT ALL PRIVILEGES ON asribank.* TO 'asriuser'@'%';
FLUSH PRIVILEGES;

USE asribank;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(50) NOT NULL,
  password VARCHAR(50) NOT NULL,
  email VARCHAR(100),
  age INT,
  balance FLOAT
);
